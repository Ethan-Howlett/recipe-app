# --- Core Web Stack ---

fastapi
uvicorn[standard]           # The ASGI server to run FastAPI
pydantic-settings           # For managing API keys & DB URLs from .env files

# --- Database (PostgreSQL) ---

sqlalchemy                  # The core ORM
psycopg[binary]             # The modern, high-performance PostgreSQL driver
alembic                     # For handling database migrations (e.g., "add column")

# --- LLM / AI (LangChain) ---

langchain                   # The main LangChain library
langchain-aws               # For connecting LangChain to Amazon Bedrock (as discussed)
boto3                       # The AWS SDK, required by langchain-aws and Textract

# --- Recipe Import Features ---

recipe-scrapers             # For parsing recipe data from website URLs
httpx                       # Modern HTTP client (for calling scraping APIs or external services)
python-docx                 # For parsing recipes from Word (.docx) files

pillow                    # Uncomment when you add the "scan a photo" feature (for OCR)

# --- Utilities & Auth ---

pint                        # For unit conversion (e.g., "cups" to "grams")
passlib[bcrypt]             # For securely hashing user passwords
python-jose[cryptography]   # For creating/validating JWT tokens (for user login)
